<script>
function addReview() {
    let t = document.getElementById('reviewInput').value.trim();
    if (!t) return;
    let d = document.createElement('div');
    d.className = 'border p-2 mt-2';
    d.textContent = t;
    document.getElementById('reviewsList').appendChild(d);
    document.getElementById('reviewInput').value = '';
    new bootstrap.Modal(document.getElementById('feedbackModal')).show();
}
</script>

<script>
function sendFeedback() {
    let n = document.getElementById('fbName').value.trim();
    let e = document.getElementById('fbEmail').value.trim();
    let m = document.getElementById('fbMsg').value.trim();
    if (!n || !e || !m) return;
    document.getElementById('fbName').value = '';
    document.getElementById('fbEmail').value = '';
    document.getElementById('fbMsg').value = '';
    new bootstrap.Modal(document.getElementById('feedbackModal')).show();
}
</script>
