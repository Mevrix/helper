<?php
session_start();
$id = $_GET['id'] ?? 0;
if ($id && isset($_SESSION['cart'][$id])) unset($_SESSION['cart'][$id]);
header('Location: cart.php');
?>

<div class="container mt-4">
    <h1>Корзина</h1>
    <a href="index.php" class="btn btn-secondary">← На главную</a>
    <?php if (empty($_SESSION['cart'])): ?>
        <div class="alert alert-info">Пусто</div>
    <?php else:
        $ids = array_keys($_SESSION['cart']);
        $in = implode(',', array_fill(0, count($ids), '?'));
        $stmt = $conn->prepare("SELECT id, name, price FROM tovar WHERE id IN ($in)");
        $stmt->bind_param(str_repeat('i', count($ids)), ...$ids);
        $stmt->execute();
        $result = $stmt->get_result();
        $total = 0;
    ?>
    <table class="table table-bordered">
        <tr><th>Товар</th><th>Цена</th><th>Кол-во</th><th>Сумма</th><th></th></tr>
        <?php while ($p = $result->fetch_assoc()):
            $qty = $_SESSION['cart'][$p['id']];
            $sum = $p['price'] * $qty;
            $total += $sum;
        ?>
        <tr>
            <td><?= $p['name'] ?></td>
            <td><?= number_format($p['price']) ?> ₽</td>
            <td><?= $qty ?></td>
            <td><?= number_format($sum) ?> ₽</td>
            <td><a href="remove_from_cart.php?id=<?= $p['id'] ?>" class="btn btn-danger btn-sm">Удалить</a></td>
        </tr>
        <?php endwhile; ?>
        <tr class="table-success"><th colspan="3">Итого</th><th><?= number_format($total) ?> ₽</th><th></th></tr>
    </table>
    <?php endif; ?>

<script>
function sendReview() {
    let a = document.getElementById('login').value.trim();
    let b = document.getElementById('otziv').value.trim();
    if (a && b) {
        document.getElementById('list').innerHTML += '<div class="border bg-dark text-white p-2 mt-2">' + a + '</div>';
        document.getElementById('list').innerHTML += '<div class="border p-2 mb-4">' + b + '</div>';
        document.getElementById('login').value = '';
        document.getElementById('otziv').value = '';
    }
}
</script>
